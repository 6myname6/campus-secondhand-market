### 1.2 反射映射核心算法
```java
// 将 ResultSet 的一行映射为实体对象
// 核心流程：读取 @Table 注解获取表名 → 遍历字段 → 
//   通过 @Column 注解或自动驼峰转蛇形映射列名 → 类型安全赋值
private static <T> T mapRow(ResultSet rs, Class<T> clazz) {
    T obj = clazz.getDeclaredConstructor().newInstance();
    for (Field field : clazz.getDeclaredFields()) {
        // 1. 取 @Column 注解值，若无则自动转换：userId → user_id
        String colName = field.isAnnotationPresent(Column.class)
            ? field.getAnnotation(Column.class).value()
            : camelToSnake(field.getName());

        // 2. 根据字段类型调用对应的 rs.getXxx() 方法
        Object value = getColumnValue(rs, colName, field.getType());

        // 3. 反射设值（私有字段也通过 setAccessible 写入）
        field.setAccessible(true);
        field.set(obj, value);
    }
    return obj;
}
```

## 2. ServerHandler.java — 请求分发核心

### 2.1 函数式路由表
```java
// dispatchMap: 将 action 字符串映射到处理函数
// 设计优势：新增业务只需 put 一行，无需修改框架代码
// Function<JsonObject, Response> 接收 params JSON，返回 Response
private final Map<String, Function<JsonObject, Response>> dispatchMap = new HashMap<>();

// 在构造器中注册路由
public ServerHandler() {
    // 公开接口 — 无需认证
    dispatchMap.put("user.login", this.userService::login);
    dispatchMap.put("goods.findById", this.goodsService::findById);

    // 需认证接口 — 通过 requireAuth 包装器注入 token 校验
    dispatchMap.put("goods.publish", requireAuth(this.goodsService::publish));
    dispatchMap.put("order.create", requireAuth(this.orderService::createOrder));
}
```

### 2.2 requireAuth — 认证包装器（高阶函数）
```java
// 关键设计：高阶函数包装 — 在不修改业务代码的前提下为方法添加认证
// 流程：提取 token → AuthService.validate() → 通过则注入 userId 并执行业务
//       → 失败则直接返回 "未登录" 错误响应
private Function<JsonObject, Response> requireAuth(Function<JsonObject, Response> handler) {
    return params -> {
        String token = params.has("token") ? params.get("token").getAsString() : null;
        Long userId = authService.validate(token);
        if (userId == null) {
            return Response.fail("请先登录");
        }
        params.addProperty("_userId", userId);  // 注入 userId 供业务层使用
        return handler.apply(params);
    };
}
```

### 2.3 消息接收与分发
```java
// channelRead0: 每条消息以 \n 分隔到达
// 流程：JSON 解析 → 提取 action → dispatchMap 查找 → 执行 → 写回响应
@Override
protected void channelRead0(ChannelHandlerContext ctx, String msg) {
    JsonObject json = JsonParser.parseString(msg).getAsJsonObject();
    String action = json.get("action").getAsString();
    JsonObject params = json.getAsJsonObject("params");

    Function<JsonObject, Response> handler = dispatchMap.get(action);
    Response resp = handler != null
        ? handler.apply(params)
        : Response.fail("未知操作: " + action);

    ctx.writeAndFlush(gson.toJson(resp) + "\n");  // \n 作为消息分隔符
}
```

## 3. BaseService.java — 事务管理基类

```java
// 事务模板方法：统一管理 Connection 的获取/提交/回滚/关闭
// 使用 Lambda 回调实现"事务包围"模式，业务代码只需关心 SQL 逻辑
protected <T> T execTx(TxFunction<Connection, T> fn) {
    Connection conn = JdbcUtils.getConnection();
    try {
        conn.setAutoCommit(false);        // 关闭自动提交
        T result = fn.apply(conn);        // 执行业务 Lambda
        conn.commit();                    // 成功则提交
        return result;
    } catch (Exception e) {
        conn.rollback();                  // 异常则回滚
        throw new RuntimeException(e);
    } finally {
        conn.setAutoCommit(true);
        conn.close();
    }
}
```

## 4. OrderService.java — 创建订单（多表事务）

```java
// 创建订单涉及 3 张表操作，必须在同一事务中完成
// 任一操作失败则全部回滚，保证数据一致性
public Response createOrder(JsonObject params) {
    return Response.success(execTx(conn -> {
        Long userId = params.get("_userId").getAsLong();
        // 1. 解析订单项：[{goodsId, quantity}, ...]
        List<OrderItemRequest> items = parseItems(params);

        // 2. 计算总金额：Σ(商品单价 × 数量)
        BigDecimal total = BigDecimal.ZERO;
        for (OrderItemRequest item : items) {
            Goods goods = JdbcUtils.queryOne(Goods.class,
                "SELECT * FROM goods WHERE goods_id = ?", item.goodsId);
            total = total.add(goods.getPrice().multiply(new BigDecimal(item.quantity)));
        }

        // 3. 生成唯一订单号：时间戳 + 随机数
        String orderNo = generateOrderNo();

        // 4. INSERT INTO orders（含收件人信息）
        Order order = new Order();
        order.setOrderNo(orderNo);
        // ... set 其他字段
        JdbcUtils.insert(conn, order);  // 传入 conn 确保在同一事务

        // 5. 批量 INSERT INTO order_items
        for (OrderItemRequest item : items) {
            OrderItem oi = new OrderItem();
            // ... set 字段
            JdbcUtils.insert(conn, oi);
        }

        // 6. UPDATE goods SET status='sold' 标记商品已售
        JdbcUtils.update(conn, "UPDATE goods SET status='sold' WHERE goods_id IN (...)");

        return order.getOrderId();
    }));
}
```

## 5. AuthService.java — Token 认证

```java
// 基于内存 ConcurrentHashMap 的 Token 管理
// ConcurrentHashMap 保证多线程并发安全
// 一个用户只保留一个活跃 Token（后续登录会挤掉之前的）
public class AuthService {
    // token → userId 映射
    private final Map<String, Long> tokenToUser = new ConcurrentHashMap<>();
    // userId → token 映射（反向索引，用于快速查找和清理旧 Token）
    private final Map<Long, String> userToToken = new ConcurrentHashMap<>();

    public String login(Long userId) {
        // 如果已有 Token，先清除旧的（互踢逻辑）
        String oldToken = userToToken.remove(userId);
        if (oldToken != null) tokenToUser.remove(oldToken);

        // 生成新 Token
        String token = UUID.randomUUID().toString();
        tokenToUser.put(token, userId);
        userToToken.put(userId, token);
        return token;
    }

    public Long validate(String token) {
        return token != null ? tokenToUser.get(token) : null;
    }

    public void logout(String token) {
        Long userId = tokenToUser.remove(token);
        if (userId != null) userToToken.remove(userId);
    }
}
```

## 6. ApiClient.java — 客户端 API 门面

```java
// 对所有服务端 action 进行类型化封装
// 每个方法对应一个服务端 action，隐藏底层 Netty 通信细节
public class ApiClient {
    private NettyClient nettyClient;

    // 底层通用请求方法
    private JsonObject send(String action, JsonObject params) {
        String json = nettyClient.send(action, params);
        return JsonParser.parseString(json).getAsJsonObject();
    }

    // 类型化的业务方法示例
    public Goods getGoodsDetail(long goodsId) {
        JsonObject params = new JsonObject();
        params.addProperty("goodsId", goodsId);
        JsonObject resp = send("goods.findById", params);
        return gson.fromJson(resp.getAsJsonObject("data"), Goods.class);
    }

    public String login(String username, String password) {
        JsonObject params = new JsonObject();
        params.addProperty("username", username);
        params.addProperty("password", password);
        JsonObject resp = send("user.login", params);
        return resp.getAsJsonObject("data").get("token").getAsString();
    }
}
```

## 7. NettyClient.java — 异步 TCP 通信

```java
// 使用 CompletableFuture 实现异步请求-响应模型
// 发送请求后阻塞等待，直到收到响应（10秒超时）
public class NettyClient {
    private Channel channel;

    public String send(String action, JsonObject params) {
        // 1. 构造 JSON 请求
        JsonObject request = new JsonObject();
        request.addProperty("action", action);
        request.add("params", params);
        request.addProperty("_reqId", reqIdCounter.incrementAndGet());

        // 2. 创建临时 Handler 和 CompletableFuture
        ClientHandler handler = new ClientHandler();
        CompletableFuture<String> future = handler.getFuture();

        // 3. pipeline 中添加 handler 并发消息
        channel.pipeline().addLast(handler);
        channel.writeAndFlush(gson.toJson(request) + "\n");

        // 4. 阻塞等待响应（10秒超时）
        String result = future.get(10, TimeUnit.SECONDS);

        // 5. 移除临时 handler
        channel.pipeline().remove(handler);
        return result;
    }
}
```

## 8. ChatView.java — 聊天界面核心逻辑

### 8.1 消息气泡渲染
```java
// 消息气泡 UI：自己（右对齐蓝色） vs 对方（左对齐灰色）
// 使用 HTML 转义防止 XSS 注入攻击
private void addMessageBubble(Message msg) {
    boolean isMine = msg.getSenderId() == AppContext.getInstance().getUserId();

    JPanel bubble = new JPanel(new FlowLayout(
        isMine ? FlowLayout.RIGHT : FlowLayout.LEFT));

    String escaped = escapeHtml(msg.getContent());  // < → &lt;   > → &gt;
    JLabel label = new JLabel("<html><div style='padding:8px; max-width:250px'>"
        + escaped + "</div></html>");

    // 自己的消息：蓝色背景 + 白色文字
    // 对方的消息：灰色背景 + 黑色文字
    label.setBackground(isMine ? new Color(70, 130, 255) : Color.LIGHT_GRAY);
    label.setOpaque(true);

    bubble.add(label);
    messagePanel.add(bubble);
    messagePanel.revalidate();
    messagePanel.repaint();
}
```

### 8.3 HTML 转义防 XSS
```java
// 聊天消息必须转义特殊字符，防止恶意用户注入 HTML/JS
// <script>alert('xss')</script> → &lt;script&gt;alert('xss')&lt;/script&gt;
private String escapeHtml(String text) {
    return text
        .replace("&", "&amp;")
        .replace("<", "&lt;")
        .replace(">", "&gt;")
        .replace("\"", "&quot;")
        .replace("'", "&#39;");
}
```

## 9. 关键设计模式总结

| 设计模式 | 应用位置 | 说明 |
|----------|----------|------|
| **模板方法** | `BaseService.execTx()` | 定义事务骨架，业务 Lambda 填充具体逻辑 |
| **高阶函数** | `ServerHandler.requireAuth()` | 参数化认证逻辑，不侵入业务代码 |
| **门面模式** | `ApiClient.java` | 封装 Netty 通信细节，对外提供类型化 API |
| **单例模式** | `AppContext.java` | 全局唯一的用户会话状态 |
| **观察者/轮询** | `ChatView.java` Timer | 定时拉取新消息 |
| **DTO** | `Request.java` / `Response.java` | 协议层与业务层数据隔离 |
| **ORM 反射映射** | `JdbcUtils.mapRow()` | 注解驱动的自动 ORM |
